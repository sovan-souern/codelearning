1. ការណែនាំអំពី JavaScript

JavaScript ជាភាសាសរសេរកម្មវិធីសម្រាប់បន្ថែមអន្តរកម្មទៅក្នុងគេហទំព័រ។

វាដំណើរការនៅលើកម្មវិធីរុករក និងអាចប្រើនៅខាងម៉ាស៊ីនមេជាមួយ Node.js។

1.1. សេចក្តីពិពណ៌នាអំពី JavaScript

JavaScript អនុញ្ញាតឱ្យបន្ថែមភាពសកម្មដូចជា ប៊ូតុង ទម្រង់ និងអន្តរកម្ម។

វាជាភាសាដែលមានភាពបត់បែន និងគាំទ្រដោយកម្ម

1.2. ប្រវត្តិសង្ខេប JavaScript

JavaScript ត្រូវបានបង្កើតឡើងដោយ Brendan Eich ក្នុងឆ្នាំ 1995។

វាបានវិវត្តទៅជា ECMAScript ជាមួយនឹងកំណែទំនើបដូចជា ES6 និងលើសពីនេះ។

1.3. ការរៀបចំបរិស្ថាន

ប្រើស្លាក <script> ដើម្បីបញ្ចូល JavaScript ក្នុង HTML។

អាចសរសេរនៅក្នុងឯកសារ .js ឬក្នុងកម្មវិធីរុករក Console។

<script>
    alert('សួស្តី JavaScript!');
</script>

សាកល្បង៖

បង្កើតទំព័រដែលបង្ហាញប្រអប់ផ្ទាំង "សួស្តី!"

2. អថេរ

អថេរត្រូវបានប្រកាសដោយ var, let, ឬ const។

let និង const មានវិសាលភាពប្លុក។

let name = 'ខ្មែរ';
const year = 2025;

2.1. ប្រភេទទិន្នន័យ

JavaScript មានប្រភេទទិន្នន័យដូចជា String, Number, Boolean, Array, Object។

let text = 'អត្ថបទ';
let number = 42;
let isTrue = true;

2.2. ប្រតិបត្តិករ

ប្រតិបត្តិករមានដូចជា +, -, *, /, ===។

let sum = 5 + 3; // 8
let isEqual = 5 === '5'; // false

សាកល្បង៖

បង្កើតអថេរដែលគណនាផលបូកនៃលេខពីរ និងបង្ហាញលទ្ធផល។

3. អនុគមន៍

អនុគមន៍ជាប្លុកកូដដែលអាចប្រើឡើងវិញបាន។

ប្រើ function ឬ Arrow Functions។

function greet(name) {
    return 'សួស្តី ' + name;
}
const add = (a, b) => a + b;

សាកល្បង៖

បង្កើតអនុគមន៍ដែលស្វាគមន៍ឈ្មោះអ្នកប្រើ។

4. លក្ខខណ្ឌ

ប្រើ if, else if, else សម្រាប់សម្រេចចិត្ត។

let age = 20;
if (age >= 18) {
    console.log('អ្នកជាមនុស្សពេញវ័យ');
} else {
    console.log('អ្នកនៅក្មេង');
}

5. រង្វិលជុំ

ប្រើ for, while, ឬ do...while សម្រាប់ធ្វើសកម្មភាពម្តងហើយម្តងទៀត។

for (let i = 0; i < 5; i++) {
    console.log(i);
}

សាកល្បង៖

បង្កើតរង្វិលជុំដែលបង្ហាញលេខពី 1 ដល់ 5។

6. អារេ

អារេជាបញ្ជីទិន្នន័យដែលអាចផ្ទុកធាតុជាច្រើន។

let fruits = ['ផ្លែប៉ោម', 'ចេក', 'ស្វាយ'];
fruits.push('ទំពាំងបាយជូរ');

7. វត្ថុ

វត្ថុផ្ទុកទិន្នន័យជាគូ key-value។

let person = {
    name: 'សុខ',
    age: 25
};
console.log(person.name);

សាកល្បង៖

បង្កើតវត្ថុដែលផ្ទុកឈ្មោះនិងអាយុ ហើយបង្ហាញឈ្មោះ។

8. ការគ្រប់គ្រប់ DOM

DOM (Document Object Model) អនុញ្ញាតឱ្យផ្លាស់ប្តូរខ្លឹមសារទំព័រ។

document.getElementById('myId').textContent = 'អត្ថបទថ្មី';

សាកល្បង៖

ផ្លាស់ប្តូរអត្ថបទនៃធាតុ HTML ដោយប្រើ DOM។

9. ព្រឹត្តិការណ៍

ព្រឹត្តិការណ៍ដូចជា click, mouseover ធ្វើឱ្យទំព័រមានអន្តរកម្ម។

document.getElementById('myButton').addEventListener('click', () => {
    alert('ប៊ូតុងត្រូវបានចុច!');
});

សាកល្បង៖

បង្កើតប៊ូតុងដែលបង្ហាញប្រអប់ផ្ទាំងនៅពេលចុច។

10. ការងារអសមកាល

JavaScript គាំទ្រការងារអសមកាលដើម្បីបង្កើតកម្មវិធីដែលមានប្រសិទ្ធភាព។

14. JavaScript Classes

Classes ជាប្រភេទទិន្នន័យដែលអាចប្រើសម្រាប់បង្កើតវត្ថុ។

ប្រើ class ដើម្បីប្រកាស និង constructor សម្រាប់កំណត់តម្លៃដើម។

class Person {
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }
    greet() {
        return `សួស្តី, ខ្ញុំឈ្មោះ ${this.name}។`;
    }
}
const person = new Person('សុខ', 25);
console.log(person.greet());

សាកល្បង៖

បង្កើត Class ដែលមានឈ្មោះ និងអាយុ ហើយបង្ហាញសារស្វាគមន៍។

15. JavaScript Fetch API

Fetch API អាចប្រើសម្រាប់ទាញយកទិន្នន័យពី API។

ប្រើ fetch() ដើម្បីទាញយកទិន្នន័យ និង .then() ដើម្បីដំណើរការវា។

fetch('https://jsonplaceholder.typicode.com/posts/1')
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error('Error:', error));

សាកល្បង៖

ប្រើ Fetch API ដើម្បីទាញយកទិន្នន័យពី API និងបង្ហាញលទ្ធផល។
