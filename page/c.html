1. ការណែនាំអំពីភាសា C

ភាសា C ជាភាសាសរសេរកម្មវិធីសម្រាប់បង្កើតប្រព័ន្ធ កម្មវិធី និងហ្គេម។

វាសាមញ្ញ បត់បែន និងគ្រប់គ្រងធនធានផ្នែករឹងបានល្អ។

1.1. សេចក្តីពិពណ៌នាអំពី C

ភាសា C គឺជាភាសារចនាសម្ព័ន្ធសម្រាប់កម្មវិធីគ្រប់ប្រភេទ។

ប្រើសម្រាប់ប្រព័ន្ធប្រតិបត្តិការ មូលដ្ឋានទិន្នន័យ និងឧបករណ៍។

1.2. ប្រវត្តិសង្ខេប C

ភាសា C បង្កើតដោយ Dennis Ritchie នៅឆ្នាំ 1972។

ប្រើដំបូងសម្រាប់ប្រព័ន្ធ Unix នៅ Bell Labs។

1.3. លក្ខណៈពិសេសនៃ C

ភាសា C សាមញ្ញ លឿន និងគ្រប់គ្រងអង្គចងចាំបាន។

អាចប្រើបានសម្រាប់កម្មវិធីគ្រប់ប្រភេទ។

1.4. ការរៀបចំបរិស្ថាន C

ត្រូវការកម្មវិធីចងក្រង (GCC) និងនិពន្ធ (VS Code)។

ដំឡើង MinGW សម្រាប់ Windows ឬ GCC សម្រាប់ Linux។

#include 
int main() {
    printf("សួស្តី ពិភពលោក!\n");
    return 0;
}

1.5. រចនាសម្ព័ន្ធកម្មវិធី C

កម្មវិធី C មានបណ្ណាល័យ អនុគមន៍ main និងសេចក្តីណែនាំ។

ឧទាហរណ៍៖ បង្ហាញសារ "សួស្តី ពិភពលោក!"។

សាកល្បង៖

បង្ហាញសារ "ខ្ញុំរៀន C!"។

2. អថេរនិងប្រភេទទិន្នន័យ

អថេរជាទីតាំងផ្ទុកទិន្នន័យនៅក្នុងអង្គចងចាំ។

ត្រូវការឈ្មោះនិងប្រភេទទិន្នន័យ។

2.1. ការប្រកាសអថេរ

ប្រកាសអថេរដោយប្រើ៖ ប្រភេទ ឈ្មោះ;

int age;
float salary;

2.2. ប្រភេទទិន្នន័យជាមូលដ្ឋាន

រួមមាន int, float, char, double។

ឧទាហរណ៍៖ int age = 20;

2.3. ថេរ

ថេរមិនអាចផ្លាស់ប្តូរបាន ប្រើ const។

const int MAX = 100;

2.4. ការបម្លែងប្រភេទ

បម្លែងប្រភេទទិន្នន័យដោយប្រើ (ប្រភេទ)។

float b = (float)5 / 2; // b = 2.5

2.5. ថ្នាក់ផ្ទុក

កំណត់វិសាលភាពអថេរ៖ auto, static, extern។

ឧទាហរណ៍៖ static int count;

សាកល្បង៖

បញ្ចូលនិងបង្ហាញអាយុ។

3. រចនាសម្ព័ន្ធគ្រប់គ្រង

គ្រប់គ្រងលំហូរកម្មវិធីដោយប្រើ if, switch, វដ្ត។

3.1. if, else

សម្រេចចិត្តដោយប្រើ if និង else។

if (age >= 18)
    printf("មនុស្សពេញវ័យ\n");
else
    printf("ក្មេង\n");

3.2. switch

ជ្រើសរើសជម្រើសច្រើនដោយ switch។

switch (grade) {
    case 'A': printf("ល្អណាស់\n"); break;
}

3.3. វដ្ត (Loops)

ប្រើ for, while សម្រាប់ការធ្វើម្តងហើយម្តងទៀត។

for (int i = 0; i < 5; i++)
    printf("%d\n", i);

4. អនុគមន៍

អនុគមន៍ជាប្លុកកូដសម្រាប់ការងារជាក់លាក់។

int add(int a, int b) {
    return a + b;
}

5. អារេ

អារេផ្ទុកទិន្នន័យច្រើននៃប្រភេទដូចគ្នា។

int numbers[5] = {1, 2, 3, 4, 5};

6. ព្រីនធឺ

ព្រីនធឺផ្ទុកអាសយដ្ឋានអង្គចងចាំ។

int x = 10;
int *p = &x;

7. ខ្សែអក្សរ

ខ្សែអក្សរជាអារេនៃតួអក្សរ។

char name[] = "Khmer";

8. ការបញ្ចូល/បញ្ចេញឯកសារ

អាននិងសរសេរឯកសារដោយប្រើ FILE។

FILE *f = fopen("data.txt", "w");
fprintf(f, "Hello");
fclose(f);
