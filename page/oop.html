1. ការណែនាំអំពី OOP

ការសរសេរកម្មវិធីបែបវត្ថុនិយម (OOP) ជួយរៀបចំកូដដោយប្រើថ្នាក់និងវត្ថុ។

វាផ្តល់នូវរចនាសម្ព័ន្ធ និងភាពអាចប្រើឡើងវិញបាន។

2. ថ្នាក់ (Classes)

ថ្នាក់គឺជាប្លង់សម្រាប់បង្កើតវត្ថុ។

ប្រើពាក្យគន្លឹះ class ដើម្បីកំណត់ថ្នាក់។

class Person:
    pass

3. វត្ថុ (Objects)

វត្ថុគឺជាការបង្កើតពីថ្នាក់។

ឧទាហរណ៍ បង្កើតវត្ថុនៃថ្នាក់ Person។

class Person:
    pass

person1 = Person()
print(person1)

សាកល្បង៖

បង្កើតវត្ថុពីថ្នាក់ Person និងបោះពុម្ពវា។

4. គុណលក្ខណៈ (Attributes)

គុណលក្ខណៈគឺជាទិន្នន័យរបស់វត្ថុ។

កំណត់គុណលក្ខណៈដោយប្រើ __init__។

class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

person1 = Person("Sokha", 25)
print(person1.name)

សាកល្បង៖

បង្កើតវត្ថុជាមួយគុណលក្ខណៈ និងបោះពុម្ពឈ្មោះ។

5. វិធីសាស្ត្រ (Methods)

វិធីសាស្ត្រគឺជាអនុគមន៍របស់ថ្នាក់។

ឧទាហរណ៍ បង្កើតវិធីសាស្ត្រសម្រាប់សួស្តី។

class Person:
    def __init__(self, name):
        self.name = name
    
    def greet(self):
        return f"សួស្តី, ខ្ញុំឈ្មោះ {self.name}!"

person1 = Person("Sokha")
print(person1.greet())

សាកល្បង៖

បង្កើតវិធីសាស្ត្រសួស្តីនិងហៅវា។

6. ការទទួលមរតក (Inheritance)

ការទទួលមរតកអនុញ្ញាតឱ្យថ្នាក់មួយទទួលគុណលក្ខណៈនិងវិធីសាស្ត្រពីថ្នាក់ផ្សេង។

ឧទាហរណ៍ ថ្នាក់ Student ទទួលមរតកពី Person។

class Person:
    def __init__(self, name):
        self.name = name
    
    def greet(self):
        return f"សួស្តី, ខ្ញុំឈ្មោះ {self.name}!"

class Student(Person):
    def __init__(self, name, grade):
        super().__init__(name)
        self.grade = grade

student1 = Student("Sokha", "A")
print(student1.greet())
print(student1.grade)

7. ពហុរូបភាព (Polymorphism)

ពហុរូបភាពអនុញ្ញាតឱ្យវត្ថុផ្សេងគ្នាឆ្លើយតបទៅនឹងវិធីសាស្ត្រដូចគ្នា។

ឧទាហរណ៍ វិធីសាស្ត្រផ្សេងគ្នាសម្រាប់ថ្នាក់ផ្សេងៗ។

class Animal:
    def speak(self):
        pass

class Dog(Animal):
    def speak(self):
        return "វ៉ុហ្វ!"

class Cat(Animal):
    def speak(self):
        return "ម៉ាវ!"

dog = Dog()
cat = Cat()
print(dog.speak())
print(cat.speak())

សាកល្បង៖

បង្កើតថ្នាក់ Animal និងថ្នាក់កូនសម្រាប់ពហុរូបភាព។

8. ការវេចខ្ចប់ (Encapsulation)

ការវេចខ្ចប់លាក់ទិន្នន័យឯកជនដោយប្រើ __។

ឧទាហរណ៍ គ្រប់គ្រងទិន្នន័យឯកជន។

class Person:
    def __init__(self, name, age):
        self.__name = name
        self.__age = age
    
    def get_name(self):
        return self.__name

person1 = Person("Sokha", 25)
print(person1.get_name())

9. ការអរូបីករណ៍ (Abstraction)

ការអរូបីករណ៍បង្ហាញតែព័ត៌មានចាំបាច់ប៉ុណ្ណោះ។

ប្រើ abc module ស democracia 

from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self):
        pass

class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius
    
    def area(self):
        return 3.14 * self.radius * self.radius

circle = Circle(5)
print(circle.area())

10. ការអនុវត្តល្អបំផុត

ប្រើឈ្មោះថ្នាក់និងវិធីសាស្ត្រដែលមានន័យ។

អនុវត្តការវេចខ្ចប់ និងរៀបចំកូដឱ្យស្អាត។

# ឧទាហរណ៍ថ្នាក់ស្អាត
class Employee:
    def __init__(self, name, salary):
        self.__name = name
        self.__salary = salary
    
    def get_info(self):
        return f"ឈ្មោះ: {self.__name}, ប្រាក់ខែ: {self.__salary}"

employee = Employee("Sokha", 50000)
print(employee.get_info())
