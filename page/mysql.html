1. ការណែនាំអំពី MySQL

MySQL ជាប្រព័ន្ធគ្រប់គ្រងមូលដ្ឋានទិន្នន័យទំនាក់ទំនង (Relational Database Management System)។

វាប្រើ SQL (Structured Query Language) ដើម្បីគ្រប់គ្រងទិន្នន័យ។

1.1. សេចក្តីពិពណ៌នាអំពី MySQL

MySQL មានល្បឿនលឿន និងអាចប្រើបានសម្រាប់គម្រោងតូចធំ។

វាត្រូវបានប្រើជាទូទៅក្នុងការអភិវឌ្ឍគេហទំព័រ។

1.2. ការដំឡើង MySQL

ទាញយក MySQL ពី mysql.com។

ដំឡើងនៅលើ Windows, Mac, ឬ Linux ហើយពិនិត្យកំណែ។

SELECT VERSION();

សាកល្បង៖

ពិនិត្យកំណែ MySQL។

2. បង្កើតនិងគ្រប់គ្រងមូលដ្ឋានទិន្នន័យ

បង្កើតមូលដ្ឋានទិន្នន័យដើម្បីរក្សាទុកទិន្នន័យ។

ប្រើ CREATE DATABASE និង USE។

CREATE DATABASE school;
USE school;

សាកល្បង៖

បង្កើតមូលដ្ឋានទិន្នន័យថ្មី។

3. បង្កើតនិងគ្រប់គ្រងតារាង

តារាងរក្សាទុកទិន្នន័យក្នុងជួរនិងជួរឈរ។

ប្រើ CREATE TABLE ដើម្បីបង្កើតតារាង។

CREATE TABLE students (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(50),
    age INT
);

4. បញ្ចូលទិន្នន័យ

បញ្ចូលទិន្នន័យទៅក្នុងតារាងដោយប្រើ INSERT INTO។

INSERT INTO students (name, age) VALUES ('Sokha', 20);
INSERT INTO students (name, age) VALUES ('Reth', 22);

សាកល្បង៖

បញ្ចូលទិន្នន័យសិស្សទៅក្នុងតារាង។

5. សួរយកទិន្នន័យ

ប្រើ SELECT ដើម្បីសួរយកទិន្នន័យពីតារាង។

ឧទាហរណ៍ ទាញយកសិស្សទាំងអស់។

SELECT * FROM students;

សាកល្បង៖

សួរយកទិន្នន័យសិស្សទាំងអស់។

6. ធ្វើបច្ចុប្បន្នភាពទិន្នន័យ

ប្រើ UPDATE ដើម្បីកែប្រែទិន្នន័យ។

ឧទាហរណ៍ ផ្លាស់ប្តូរអាយុរបស់សិស្ស។

UPDATE students SET age = 21 WHERE name = 'Sokha';

7. លុបទិន្នន័យ

ប្រើ DELETE ដើម្បីលុបទិន្នន័យចេញពីតារាង។

ឧទាហរណ៍ លុបសិស្សម្នាក់។

DELETE FROM students WHERE name = 'Reth';

8. ការភ្ជាប់តារាង (Joins)

ប្រើ JOIN ដើម្បីភ្ជាប់ទិន្នន័យពីតារាងច្រើន។

ឧទាហរណ៍ ភ្ជាប់តារាងសិស្សនិងវគ្គសិក្សា។

CREATE TABLE courses (
    course_id INT PRIMARY KEY AUTO_INCREMENT,
    course_name VARCHAR(50),
    student_id INT
);
INSERT INTO courses (course_name, student_id) VALUES ('Math', 1);
SELECT students.name, courses.course_name
FROM students
INNER JOIN courses ON students.id = courses.student_id;

9. សន្ទស្សន៍ (Indexes)

សន្ទស្សន៍បង្កើនល្បឿននៃការសួរយកទិន្នន័យ។

ប្រើ CREATE INDEX ដើម្បីបង្កើតសន្ទស្សន៍។

CREATE INDEX idx_name ON students (name);

10. ការអនុវត្តល្អបំផុត

ប្រើឈ្មោះតារាងនិងជួរឈរដែលមានន័យ។

បង្កើតសន្ទស្សន៍សម្រាប់ជួរឈរដែលសួរញឹកញាប់ និងប្រើទិន្នន័យប្រភេទត្រឹមត្រូវ។

ALTER TABLE students ADD email VARCHAR(100);
