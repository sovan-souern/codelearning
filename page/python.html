1. ការណែនាំអំពី Python

Python ជាភាសាសរសេរកម្មវិធីដែលងាយស្រួលរៀននិងប្រើប្រាស់។

វាត្រូវបានប្រើសម្រាប់ការអភិវឌ្ឍគេហទំព័រ ការវិភាគទិន្នន័យ និងអនុគមន៍សិប្បនិម្មិត។

1.1. សេចក្តីពិពណ៌នាអំពី Python

Python មានវាក្យសម្ព័ន្ធសាមញ្ញ និងអាចអានបាន។

វាគាំទ្រការសរសេរកម្មវិធីបែបវត្ថុនិយម និងមុខងារនិយម។

1.2. ការដំឡើង Python

ទាញយក Python ពី python.org។

ដំឡើងនៅលើ Windows, Mac, ឬ Linux ហើយពិនិត្យកំណែ។

# ពិនិត្យកំណែ Python
import sys
print(sys.version)

សាកល្បង៖

ពិនិត្យកំណែ Python។

2. អថេរនិងប្រភេទទិន្នន័យ

អថេរគឺជាកន្លែងរក្សាទុកទិន្នន័យ។

Python មានប្រភេទទិន្នន័យដូចជា ចំនួនគត់ អក្សរ និងបញ្ជី។

name = "Sokha"
age = 25
height = 1.75
is_student = True

សាកល្បង៖

បង្កើតអថេរនិងបោះពុម្ពវា។

3. ប្រតិបត្តិករ

ប្រតិបត្តិករប្រើសម្រាប់គណនានិងប្រៀបធៀប។

រួមមាន + , - , * , / , == , និង != ។

a = 10
b = 5
sum = a + b
is_equal = a == b

4. លក្ខខណ្ឌ

ប្រើ if, elif, else ដើម្បីសម្រេចចិត្ត។

ឧទាហរណ៍ ពិនិត្យអាយុ។

age = 18
if age >= 18:
    print("អ្នកជាមនុស្សពេញវ័យ")
else:
    print("អ្នកនៅក្មេង")

សាកល្បង៖

សរសេរលក្ខខណ្ឌដើម្បីពិនិត្យអាយុ។

5. រង្វិលជុំ

ប្រើ for និង while ដើម្បីធ្វើសកម្មភាពម្តងហើយម្តងទៀត។

ឧទាហរណ៍ បោះពុម្ពលេខ 1 ដល់ 5។

for i in range(1, 6):
    print(i)

6. អនុគមន៍

អនុគមន៍ជាបណ្តុំនៃកូដដែលអាចប្រើឡើងវិញបាន។

ប្រើ def ដើម្បីកំណត់អនុគមន៍។

def greet(name):
    return f"សួស្តី {name}!"
print(greet("Sokha"))

សាកល្បង៖

បង្កើតអនុគមន៍សម្រាប់សួស្តី។

7. បញ្ជី (Lists)

បញ្ជីជាកន្លែងរក្សាទុកទិន្នន័យជាច្រើន។

អាចបន្ថែម លុប ឬកែទិន្នន័យ។

fruits = ["ផ្លែប៉ោម", "ចេក", "ស្វាយ"]
fruits.append("ទំពាំងបាយជូ")
print(fruits)

8. វចនានុក្រម (Dictionaries)

វចនានុក្រមរក្សាទុកទិន្នន័យជាគូ key-value។

ឧទាហរណ៍ ព័ត៌មានសិស្ស។

student = {"name": "Sokha", "age": 25}
print(student["name"])

9. ម៉ូឌុលនិងកញ្ចប់

ម៉ូឌុលជួយរៀបចំកូដ។

ឧទាហរណ៍ ប្រើម៉ូឌុល math។

import math
print(math.sqrt(16))

10. ការអនុវត្តល្អបំផុត

សរសេរកូដឱ្យច្បាស់លាស់និងអាចអានបាន។

ប្រើមតិយោបល់ (comments) និងឈ្មោះអថេរដែលមានន័យ។

# គណនាផ្ទៃក្រឡាការ៉េ
side = 5
area = side * side
print(f"ផ្ទៃក្រឡាការ៉េ: {area}")
